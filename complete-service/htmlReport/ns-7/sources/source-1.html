 import lombok.extern.slf4j.Slf4j;

 import java.util.List;
 
 @Service
 @Slf4j
 public class ConsumerService {
     @Autowired
     UserRepository userRepository ;
     @Autowired
     JmsTemplate jmsTemplate ;
 
     @JmsListener(destination = "queue.user")
     public void saveUser(Message<User> object){
         try {
             userRepository.save(object.getPayload()).subscribe();
             log.info("received user message {}",object);
         } catch (Exception e) {
             e.printStackTrace();
             log.error("Error receive in activemq : {}", e.getMessage());
         }
     }
     @JmsListener(destination = "queue.update.user")
     public void updateUser(Message<User> object){
         try {
             userRepository.save(object.getPayload()).subscribe();
             log.info("received user message {}",object);
         } catch (Exception e) {
             e.printStackTrace();
             log.error("Error receive in activemq : {}", e.getMessage());
         }
     }
 
     @JmsListener(destination = "queue.user.get")
     public void getAll(Message<List<User>> message) {
         try {
             List<User> userList = message.getPayload();
             log.info("userList data : {} ", userList);
         } catch (Exception e) {
             log.error("ERROR :", e.getMessage(), e);
         }
     }
     }
